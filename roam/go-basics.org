#+title: GO Basics

* Core Basics

** Variables

#+begin_src go
package main

import "fmt"

func main() {
    // Explicit type
    var name string = "Alice"
    var age int = 25
    var balance float64 = 100.50
    var active bool = true
    
    // Type inference (Go figures it out)
    var city = "New York"  // string
    var count = 42         // int
    
    // Short declaration (inside functions only)
    message := "Hello"
    x := 10
    
    // Short declaration NOT allowed at package level
    // name := "Bob"  // ERROR outside function
    
    fmt.Println(name, age, city, message)
}
#+end_src


- Go is *strongly typed*: once a variable is declared, its type is fixed
- Zero values: strings =""=, ints =0=, bools =false=, pointers =nil=, slice =nil=

  
** Constants

Constants must be known at compile time. You cannot assign values returned from functions or computed at runtime.

#+begin_src go
// Constants (values known at compile time)
const Pi = 3.14159
const MaxConnections int = 100

// Group constants
const (
    Red    = "red"
    Green  = "green"
    Blue   = "blue"
)

// Iota (auto-incrementing)
const (
    Small  = iota  // 0
    Medium         // 1
    Large          // 2
)

// Constants can't be reassigned
// Pi = 3.15  // ERROR

** Pointers

Pointers hold memory addresses. Needed for efficiency (large structs) and modification.

#+begin_src go
x := 10
ptr := &x      // Pointer to x (address)
*ptr = 20      // Dereference: change value at pointer
fmt.Println(x) // 20

// Why: Pass large structs by pointer, modify in functions
type Server struct {
    Host string
    Port int
}

func updatePort(s *Server, port int) {
    s.Port = port  // Auto-dereferenced
}

server := Server{Host: "localhost", Port: 8080}
updatePort(&server, 9000)
fmt.Println(server.Port) // 9000
#+end_src


* Control Flow

** If/Else
You must put the opening braces on the same line as the condition and not on a new line. And =else= and =else if= must start on the same line as the closing braces of =if= statement. 

#+begin_src go
if x > 10 {
    fmt.Println("big")
} else if x > 5 {
    fmt.Println("medium")
} else {
    fmt.Println("small")
}

// If with initialization
if err := doSomething(); err != nil {
    return err
}
// err not accessible here
#+end_src

** For Loops

Go only has =for=, no =while=

#+begin_src go
// Traditional
for i := 0; i < 10; i++ {
    fmt.Println(i)
}

// While-style
for condition {
    // loop
}

// Infinite loop
for {
    // loop forever
    break  // exit
}

// Continue
for i := 0; i < 10; i++ {
    if i%2 == 0 {
        continue  // Skip even numbers
    }
    fmt.Println(i)
}
#+end_src

** Range Loop

Iterate over slices, arrays, maps, strings.

#+begin_src go
// Slice: index and value
nums := []int{10, 20, 30}
for i, val := range nums {
    fmt.Println(i, val)
}

// Ignore index
for _, val := range nums {
    fmt.Println(val)
}

// Ignore value (just indices)
for i := range nums {
    fmt.Println(i)
}

// Map: key and value
env := map[string]string{"PATH": "/usr/bin", "HOME": "/root"}
for key, val := range env {
    fmt.Println(key, "=", val)
}

// String: index and rune (Unicode code point)
for i, char := range "hello" {
    fmt.Printf("%d: %c\n", i, char)
}
#+end_src

* Functions

#+begin_src go
// Basic function
func add(a, b int) int {
    return a + b
}

// Multiple returns (idiomatic Go for error handling)
func divide(a, b int) (int, error) {
    if b == 0 {
        return 0, fmt.Errorf("divide by zero")
    }
    return a / b, nil
}

result, err := divide(10, 2)
if err != nil {
    fmt.Println("Error:", err)
    return
}

// Named return values
func split(sum int) (x, y int) {
    x = sum * 4 / 9
    y = sum - x
    return  // Returns x and y
}

// Variadic functions (variable arguments)
func sum(nums ...int) int {
    total := 0
    for _, n := range nums {
        total += n
    }
    return total
}

sum(1, 2, 3, 4, 5)  // 15
#+end_src

* Error Handling

#+begin_src go

** Defer

Defer schedules a function call to run when the surrounding function exits. Used for cleanup.

#+begin_src go
func main() {
    file, _ := os.Open("data.txt")
    defer file.Close()  // Runs when main() exits
    
    // Do stuff with file...
}

// Multiple defers execute in LIFO order (stack)
defer fmt.Println("3")
defer fmt.Println("2")
defer fmt.Println("1")  // Prints: 1, 2, 3
#+end_src

Common uses: closing files, unlocking mutexes, closing connections.

** Buffered Write

#+begin_src go
package main

import (
	"bufio"
	"os"
)

func main() {
	file, _ := os.Create("data.txt")
	defer file.Close()

	writer := bufio.NewWriter(file)

	writer.WriteString("hello\n")
	writer.WriteString("world\n")

	writer.Flush()  // Important: flush buffer to disk
}
#+end_src

Without buffer every write will call disk (slow), with buffer, it collects data and writes in chunks (fast)


* JSON

Essential for config files, APIs, data exchange.

#+begin_src go
import "encoding/json"

type Config struct {
    Host    string `json:"host"`
    Port    int    `json:"port"`
    Debug   bool   `json:"debug,omitempty"`  // Omit if false
    Timeout int    `json:"-"`                // Never serialize
}

// Parse JSON (Unmarshal)
data := []byte(`{"host":"localhost","port":8080,"debug":true}`)
var cfg Config
err := json.Unmarshal(data, &cfg)
if err != nil {
    log.Fatal(err)
}
fmt.Println(cfg.Host, cfg.Port)

// Create JSON (Marshal)
cfg := Config{Host: "0.0.0.0", Port: 3000, Debug: true}

// Compact
bytes, _ := json.Marshal(cfg)
fmt.Println(string(bytes))

// Pretty print
bytes, _ = json.MarshalIndent(cfg, "", "  ")
fmt.Println(string(bytes))

// Working with files
data, _ := os.ReadFile("config.json")
json.Unmarshal(data, &cfg)

bytes, _ := json.MarshalIndent(cfg, "", "  ")
os.WriteFile("config.json", bytes, 0644)

// Dynamic JSON (map)
var result map[string]interface{}
json.Unmarshal(data, &result)
fmt.Println(result["host"])
#+end_src
